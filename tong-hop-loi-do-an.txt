4. Query 21 và 23 có nguy cơ tính sai TOTAL_CAPACITY
- query 21 "SUM(w.`CAPACITY`) AS TOTAL_CAPACITY", Nhưng bảng WAREHOUSE đang JOIN với ORDER_WAREHOUSE_LOG. Một kho có nhiều log thì capacity của kho đó bị cộng lặp nhiều lần
- Ví dụ kho WH_HN_01 có 5 log, capacity 5000, khi JOIN sẽ có thể bị tính thành 25000, tốt hơn là tách aggregate
Mẫu sửa:
WITH warehouse_capacity AS (
    SELECT
        BRANCH_CODE,
        COUNT(*) AS TOTAL_WAREHOUSES,
        SUM(CAPACITY) AS TOTAL_CAPACITY
    FROM WAREHOUSE
    GROUP BY BRANCH_CODE
),
warehouse_activity AS (
    SELECT
        w.BRANCH_CODE,
        COUNT(owl.ORDER_CODE) AS TOTAL_ORDERS_STORED,
        COUNT(CASE WHEN owl.STORAGE_STATUS = 'In' THEN 1 END) AS ORDERS_IN,
        COUNT(CASE WHEN owl.STORAGE_STATUS = 'Out' THEN 1 END) AS ORDERS_OUT
    FROM WAREHOUSE w
    LEFT JOIN ORDER_WAREHOUSE_LOG owl
        ON w.WAREHOUSE_CODE = owl.WAREHOUSE_CODE
    GROUP BY w.BRANCH_CODE
)
SELECT
    b.BRANCH_CODE,
    b.BRANCH_NAME,
    wc.TOTAL_WAREHOUSES,
    wc.TOTAL_CAPACITY,
    wa.TOTAL_ORDERS_STORED,
    wa.ORDERS_IN,
    wa.ORDERS_OUT
FROM BRANCH b
JOIN warehouse_capacity wc ON b.BRANCH_CODE = wc.BRANCH_CODE
LEFT JOIN warehouse_activity wa ON b.BRANCH_CODE = wa.BRANCH_CODE;
